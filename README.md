### Introduction
* Currently supports linked lists and slices
* No dependencies outside the standard library

### Example
```go
package main

import (
	"fmt"

	"github.com/gofor-little/ts"
)

func main() {
    list := ts.LinkedList{}
    list.Push("Some string")

    item := list.GetTail()
    fmt.Println(item)

    item = list.Pop()
    fmt.Println(item)

    slice := ts.Slice{}

    slice.Add(1, 2, 3)
    fmt.Println(slice.GetElements())

    slice.Remove(1, 2, 3)
    fmt.Println(slice.GetElements())
}
```
